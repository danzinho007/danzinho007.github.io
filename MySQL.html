alter table pessoas
add column codigo int first;
 Adiciona a coluna codigo sendo a primeira da tabela pessoas

alter table pessoas
add column profissao varchar(10);
 Adiciona a coluna profissão na última linha da tabela pessoas

alter table pessoas
add column profissao varchar(10) after nome;
 Adiciona a coluna profissão depois da coluna nome na tabela pessoas

alter table pessoas
change column profissao prof varchar(20) not null default '';;
 Renomear a coluna profissao pra prof na tabela pessoas

alter table pessoas
drop column profissao;
 Elimina a coluna profissão da tabela pessoas

alter table pessoas
modify column profissao varchar(20) not null default '';
 Redifinir o tipo primitivo / constraints
Ao invés de ser not null agora será default
