    Obs¹ : Essas tags precisam ser colocadas dentro da tag

    -------- Letra N --------

    var n0 = window.prompt('Digite o número: ')
    Isso mostra uma janela e guarda o valor digitado na var n0

    var n1 = window.prompt('Digite o 1° número: ')
    var n2 = window.prompt('Digite o 2° número: ')
    var s1 = n1 + n2
    window.alert('String + String junta, logo ' + n1 + ' + ' + n2 + ' é ' + s1)
    Vai mostrar na tela : String + String junta, logo n1 + n2 é n1n2
    Isso irá juntar os 2 valores, pois o comando window.prompt só lê string

    var n3 = Number.parseInt(window.prompt('Digite o 3° número : '))
    Aqui estamos convertendo a string window.prompt em Número Inteiro, caso
    a pessoa digite número real, vai contar somente a parte inteira.
    Exemplo 4.50 , parte inteira é 4

    var n4 = parseInt(window.prompt('Digite o 4° número : '))
    Outra forma de converter a string em um Número Inteiro

    var n5 = window.prompt('Digite o 5° número: ')
    var n6 = Number.parseInt(window.prompt('Digite o 6° número: '))
    var s3 = n5 + n6
    window.alert('String + Numero é junta, logo '+ s3)
     Isso irá juntar os 2 valores, pois o comando window.prompt lê string
    String + Número = Junção

    var n7 = Number.parseFloat(window.prompt('Digite o 7° número : '))
    Aqui estamos convertendo a string window.prompt em número real
    Ex 7 é inteiro, 7.0 é real
    var n8 = parseFloat(window.prompt('Digite o 8° número: '))
    Outra forma de converter a string em um Número Real
    var s4 = n7 + n8
    window.alert('A soma de ' + n7 + ' e ' + n8 + ' é: ' + s4)
    Irá mostrar a soma de 2 números reais
    Se a pessoa colocar 7 e 7.7, a resposta será 14.7, pois
    convertemos os 2 númeres antes em reais / flutuantes

    var nome = window.prompt("Qual é o seu nome ?")
    O comando window.prompt só lê string
    Vai mostrar uma janela quando o usuário entrar no site e perguntar o nome
    Ao invés de usar widow.prompt pode usar let nome ou const nome
